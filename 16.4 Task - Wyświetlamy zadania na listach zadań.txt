SELECT * FROM ISSUES
JOIN ISSUESLISTS IL ON IL.ID = ISSUES.ISSUESLIST_ID
GROUP BY ISSUES.ID;


SELECT * FROM ISSUES
JOIN USERS U on ISSUES.USER_ID_ASSIGNEDTO = U.ID;


SELECT U.ID, U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) >= 1;
